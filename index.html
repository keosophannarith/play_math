ជំហានទី១៖ កូពីកូដនេះទៅដាក់ក្នុង Apps Script របស់អ្នក (ជំនួសកូដចាស់)

function doGet(e) {
  if (e.parameter.action == "getStudents") {
    var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Students");
    var data = sheet.getDataRange().getValues();
    var result = [];
    for (var i = 1; i < data.length; i++) {
      if (data[i][0] != "") {
        result.push({
          id: data[i][0].toString(),
          name: data[i][1].toString(),
          photo: data[i][2].toString(),
          pin: data[i][3] ? data[i][3].toString() : "0000",
          completed: data[i][4] === true || data[i][4] === "true"
        });
      }
    }
    return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
  }
}
